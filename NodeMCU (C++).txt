#include <ESP8266WiFi.h>
#include <FirebaseESP8266.h>
#include <PubSubClient.h>
#include <Ticker.h>
#include <math.h>

// ---------- WIFI ----------
const char* WIFI_SSID = "SUA_REDE";
const char* WIFI_PASSWORD = "SUA_SENHA";

// ---------- FIREBASE ----------
#define FIREBASE_HOST "monitor-caixa-agua-ff63a-default-rtdb.firebaseio.com"
#define FIREBASE_AUTH "SUA_API_KEY"

// ---------- MQTT ----------
const char* MQTT_BROKER = "192.168.1.10";
const int   MQTT_PORT   = 1883;
const char* MQTT_TOPIC  = "casa/caixa_agua/dados";

// ---------- PINOS ----------
#define PIN_TRIG D5
#define PIN_ECHO D6
#define PIN_RELE D1

// ---------- MEDIDAS DA CAIXA ----------
const float ALTURA_CAIXA = 76.0;
const float DISTANCIA_CHEIA = 10.0;
const float RAIO_BASE = 58.0;
const float RAIO_TOPO = 75.5;

// ---------- CONTROLE BOMBA ----------
const float NIVEL_LIGA  = 20.0;
const float NIVEL_DESLIGA = 90.0;

bool bombaLigada = false;

// ---------- OBJETOS ----------
FirebaseData firebaseData;
FirebaseConfig config;
FirebaseAuth auth;
WiFiClient espClient;
PubSubClient mqttClient(espClient);
Ticker watchdog;

// ---------- WATCHDOG ----------
void reiniciar() {
  ESP.restart();
}

// ---------- WIFI ----------
void conectarWiFi() {
  if (WiFi.status() == WL_CONNECTED) return;

  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  unsigned long inicio = millis();

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    if (millis() - inicio > 20000) ESP.restart();
  }
}

// ---------- MQTT ----------
void reconnectMQTT() {
  while (!mqttClient.connected()) {
    mqttClient.connect("ESP8266_Caixa");
    delay(2000);
  }
}

// ---------- SENSOR COM FILTRO ----------
float lerNivelFiltrado() {

  float leituras[10];
  int validas = 0;

  for (int i = 0; i < 10; i++) {

    digitalWrite(PIN_TRIG, LOW);
    delayMicroseconds(2);
    digitalWrite(PIN_TRIG, HIGH);
    delayMicroseconds(10);
    digitalWrite(PIN_TRIG, LOW);

    long duracao = pulseIn(PIN_ECHO, HIGH, 30000);
    float distancia = duracao * 0.034 / 2.0;

    if (distancia > 5 && distancia < ALTURA_CAIXA + 30) {
      leituras[validas++] = distancia;
    }

    delay(60);
  }

  if (validas < 3) return 0;

  // Média simples
  float soma = 0;
  for (int i = 0; i < validas; i++) soma += leituras[i];
  float distanciaMedia = soma / validas;

  float nivel_cm = ALTURA_CAIXA - (distanciaMedia - DISTANCIA_CHEIA);

  if (nivel_cm < 0) nivel_cm = 0;
  if (nivel_cm > ALTURA_CAIXA) nivel_cm = ALTURA_CAIXA;

  return nivel_cm;
}

// ---------- CONTROLE BOMBA ----------
void controlarBomba(float nivel_percent) {

  if (!bombaLigada && nivel_percent <= NIVEL_LIGA) {
    digitalWrite(PIN_RELE, LOW);
    bombaLigada = true;
  }

  if (bombaLigada && nivel_percent >= NIVEL_DESLIGA) {
    digitalWrite(PIN_RELE, HIGH);
    bombaLigada = false;
  }
}

// ---------- SETUP ----------
void setup() {

  Serial.begin(115200);

  pinMode(PIN_TRIG, OUTPUT);
  pinMode(PIN_ECHO, INPUT);
  pinMode(PIN_RELE, OUTPUT);
  digitalWrite(PIN_RELE, HIGH);

  conectarWiFi();

  config.host = FIREBASE_HOST;
  config.signer.tokens.legacy_token = FIREBASE_AUTH;
  Firebase.begin(&config, &auth);

  mqttClient.setServer(MQTT_BROKER, MQTT_PORT);

  watchdog.attach(120, reiniciar);
}

// ---------- LOOP ----------
void loop() {

  watchdog.detach();
  watchdog.attach(120, reiniciar);

  conectarWiFi();
  if (!mqttClient.connected()) reconnectMQTT();
  mqttClient.loop();

  float nivel_cm = lerNivelFiltrado();
  float nivel_percent = (nivel_cm / ALTURA_CAIXA) * 100.0;

  // -------- TRONCO DE CONE --------
  float raioAtual = RAIO_BASE +
                    (RAIO_TOPO - RAIO_BASE) *
                    (nivel_cm / ALTURA_CAIXA);

  float volume_cm3 =
      (3.14159 * nivel_cm / 3.0) *
      (pow(raioAtual,2) +
       (raioAtual * RAIO_BASE) +
       pow(RAIO_BASE,2));

  float litros = volume_cm3 / 1000.0;

  controlarBomba(nivel_percent);

  // -------- FIREBASE --------
  Firebase.setFloat(firebaseData, "/nivel", nivel_percent);
  Firebase.setFloat(firebaseData, "/litros", litros);
  Firebase.setBool(firebaseData, "/bomba", bombaLigada);

  // -------- MQTT --------
  String payload = "{";
  payload += "\"nivel\":" + String(nivel_percent,1) + ",";
  payload += "\"litros\":" + String(litros,0) + ",";
  payload += "\"bomba\":" + String(bombaLigada ? "true" : "false");
  payload += "}";

  mqttClient.publish(MQTT_TOPIC, payload.c_str());

  Serial.println(payload);

  delay(5000);
}
